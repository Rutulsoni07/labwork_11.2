For example,
Input:
Enter the array's size: 5

Enter array elements:
a[0] = 5
a[1] = 9
a[2] = 4
a[3] = 7
a[4] = 3


Output:
Reversed array elements:
3, 7, 4, 9, 5

        printf("a[%d] = ", i);
        scanf("%d", &arr[i]);
    }


    int i;
    int *p,**q,***r;
    p=&arr[size-1];
    q=&p;
    r=&q;
    printf("Reversed array elements:");
    for(i=0;i<size;i++){
        printf("%d,",***r);
        p--;
    }
    return 0;
}
